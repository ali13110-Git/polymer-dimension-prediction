export default {
  async fetch(request, env) {
    const url = new URL(request.url);

    // 1. ROUTE: API Requests (must start with /check-dimension)
    if (url.pathname.startsWith("/check-dimension/train") || url.pathname.startsWith("/check-dimension/predict")) {
      // We strip "/check-dimension" so the Pi sees a clean "/train" or "/predict"
      const targetPath = url.pathname.replace("/check-dimension", "");
      const piUrl = "https://check-dimension.do-innovation.com" + targetPath;

      try {
        const response = await fetch(new Request(piUrl, request));
        // Ensure CORS headers are present for the response
        const newHeaders = new Headers(response.headers);
        newHeaders.set("Access-Control-Allow-Origin", "*");
        return new Response(response.body, {
          status: response.status,
          statusText: response.statusText,
          headers: newHeaders
        });
      } catch (e) {
        return new Response(JSON.stringify({detail: "Worker reached, but Pi Tunnel is not responding."}), {
          status: 502,
          headers: { "Content-Type": "application/json", "Access-Control-Allow-Origin": "*" }
        });
      }
    }

    // 2. ROUTE: The Website (index.html)
    // If they visit the main tool page
    if (url.pathname === "/check-dimension" || url.pathname === "/check-dimension/") {
      const site = await env.MY_BUCKET.get("index.html");
      if (site) {
        return new Response(site.body, { headers: { "Content-Type": "text/html" } });
      }
    }

    // 3. FALLBACK: Let WordPress handle everything else
    return fetch(request);
  },
};
